# RAILBIRD Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a read-only form guide & edge finder for Gigling Racing (Gigaverse) that infers hidden pet stats from race history and highlights +EV races against the payout structure.

**Architecture:** Next.js 14 App Router with server-side route proxies for all external API calls (CORS avoidance + caching). Model runs in TypeScript on the server: ELO + rarity prior → softmax win probabilities → fair odds → EV vs previewPayouts. Real-time via Pusher public channels. No auth, no writes.

**Tech Stack:** Next.js 14 (App Router), TypeScript 5, Tailwind CSS 3, viem (Abstract mainnet), pusher-js, @tanstack/react-query, Vercel deployment.

**Deadline:** 26 June 2026 (GIGATHON #1 submission).

// lib/encode.ts
import type { Faction, ItemsMode, Weather } from '@/types/racing'

// extraParamIds encoding per ТЗ §2.5
const ITEMS_MAP: Record<number, ItemsMode> = { 0: 'none', 1: 'dung', 2: 'butterflies', 3: 'all' }
const WEATHER_MAP: Record<number, Weather> = { 0: 'hot', 1: 'cold', 2: 'rainy', 3: 'snowing' }
const FACTION_MAP: Record<number, Faction> = {
  0: 'none', 1: 'crusader', 2: 'overseer', 3: 'athena',
  4: 'archon', 5: 'foxglove', 6: 'summoner', 7: 'chobo', 8: 'gigus',
}

export function decodeExtraParams(ids: number[], vals: number[]): {
  itemsMode: ItemsMode; weather: Weather | null; faction: Faction | null
} {
  let itemsMode: ItemsMode = 'none'
  let weather: Weather | null = null
  let faction: Faction | null = null

  for (let i = 0; i < ids.length; i++) {
    const id = ids[i]; const val = vals[i]
    if (id === 100) itemsMode = ITEMS_MAP[val] ?? 'none'
    else if (id === 200) weather = WEATHER_MAP[val] ?? null
    else if (id === 300) faction = FACTION_MAP[val] ?? null
  }
  return { itemsMode, weather, faction }
}

export function rarityOrdinal(r: string): number {
  return { uncommon: 0, rare: 1, epic: 2, legendary: 3, relic: 4, giga: 5 }[r.toLowerCase()] ?? 0
}

export function normalizeRarity(r: string): import('@/types/racing').Rarity {
  const map: Record<string, import('@/types/racing').Rarity> = {
    uncommon: 'uncommon', rare: 'rare', epic: 'epic',
    legendary: 'legendary', relic: 'relic', giga: 'giga',
  }
  return map[r.toLowerCase()] ?? 'uncommon'
}

export function normalizeFaction(f: string): Faction {
  const map: Record<string, Faction> = {
    none: 'none', crusader: 'crusader', overseer: 'overseer', athena: 'athena',
    archon: 'archon', foxglove: 'foxglove', summoner: 'summoner', chobo: 'chobo', gigus: 'gigus',
  }
  return map[f?.toLowerCase()] ?? 'none'
}

export function weiToEth(wei: string): number {
  return Number(BigInt(wei || '0')) / 1e18
}

export function distanceBucket(trackLength: number): 'short' | 'medium' | 'long' {
  if (trackLength <= 400) return 'short'
  if (trackLength <= 800) return 'medium'
  return 'long'
}

import type {
  ApiRaceSummary, ApiRaceDetail, ApiPetStats, ApiLeaderboardEntry,
} from '@/types/racing'

const BASE = 'https://gigaverse.io/api/racing'

async function apiFetch<T>(path: string, init?: RequestInit): Promise<T> {
  const res = await fetch(`${BASE}${path}`, {
    ...init,
    next: { revalidate: 60 }, // Next.js cache tag
  })
  if (!res.ok) throw new Error(`Gigaverse API ${path} → ${res.status}`)
  return res.json() as Promise<T>
}

export async function fetchRaces(limit = 50): Promise<ApiRaceSummary[]> {
  const data = await apiFetch<ApiRaceSummary[] | { races: ApiRaceSummary[] }>(`/races?limit=${limit}`)
  return Array.isArray(data) ? data : data.races ?? []
}

export async function fetchRace(raceId: string): Promise<ApiRaceDetail> {
  return apiFetch<ApiRaceDetail>(`/race/${raceId}`)
}

export async function fetchLobbySync(): Promise<{ races: ApiRaceSummary[] }> {
  return apiFetch<{ races: ApiRaceSummary[] }>('/lobby/sync', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({}),
    next: { revalidate: 30 },
  })
}

export async function fetchPetStats(petId: number): Promise<ApiPetStats> {
  return apiFetch<ApiPetStats>(`/pets/${petId}/stats`)
}

export async function fetchPetsStatsBatch(petIds: number[]): Promise<ApiPetStats[]> {
  const data = await apiFetch<ApiPetStats[] | Record<string, ApiPetStats>>(
    `/pets/stats?ids=${petIds.join(',')}`
  )
  return Array.isArray(data) ? data : Object.values(data)
}

export async function fetchLeaderboard(params?: {
  limit?: number
  offset?: number
  factions?: string
  rarities?: string
  genders?: string
}): Promise<{ entries: ApiLeaderboardEntry[]; total: number }> {
  const q = new URLSearchParams()
  if (params?.limit) q.set('limit', String(params.limit))
  if (params?.offset) q.set('offset', String(params.offset))
  if (params?.factions) q.set('factions', params.factions)
  if (params?.rarities) q.set('rarities', params.rarities)
  if (params?.genders) q.set('genders', params.genders)
  const data = await apiFetch<{ entries: ApiLeaderboardEntry[] } | ApiLeaderboardEntry[]>(
    `/leaderboard/elo?${q}`
  )
  if (Array.isArray(data)) return { entries: data, total: data.length }
  return { entries: data.entries ?? [], total: (data as any).total ?? 0 }
}

export async function fetchScheduled(): Promise<ApiRaceSummary[]> {
  const data = await apiFetch<ApiRaceSummary[] | { races: ApiRaceSummary[] }>('/scheduled')
  return Array.isArray(data) ? data : data.races ?? []
}

export async function fetchGlobalStats(): Promise<Record<string, unknown>> {
  return apiFetch('/stats')
}

// lib/chain/petRacing.ts
import { createPublicClient, http, parseAbi } from 'viem'
import { abstract } from 'viem/chains'
import { PET_RACING_ABI } from './abi'
import type { PayoutPreview } from '@/types/racing'
import { weiToEth } from '@/lib/encode'

const CONTRACT = '0x16e0B3D6394CE7597D34b73f5E5Fb165fD74394E' as const

function getClient() {
  return createPublicClient({
    chain: abstract,
    transport: http(process.env.ABSTRACT_RPC_URL ?? 'https://api.mainnet.abs.xyz'),
  })
}

export async function previewPayouts(raceId: string): Promise<PayoutPreview | null> {
  try {
    const client = getClient()
    const [payoutsRaw, prizePoolRaw] = await client.readContract({
      address: CONTRACT,
      abi: PET_RACING_ABI,
      functionName: 'previewPayouts',
      args: [BigInt(raceId)],
    }) as [bigint[], bigint]

    const raceData = await client.readContract({
      address: CONTRACT,
      abi: PET_RACING_ABI,
      functionName: 'getRace',
      args: [BigInt(raceId)],
    }) as {
      phase: number; entryFee: bigint; prizePool: bigint; fieldSize: bigint;
      filledSlots: bigint; trackLength: bigint; extraParamIds: bigint[]; extraParamVals: bigint[]
    }

    return {
      entryFee: raceData.entryFee.toString(),
      fieldSize: Number(raceData.fieldSize),
      prizePool: prizePoolRaw.toString(),
      payouts: payoutsRaw.map((amt, i) => ({ rank: i + 1, amount: amt.toString() })),
      jackpotEligible: raceData.entryFee > 0n,
    }
  } catch (e) {
    console.error('previewPayouts failed:', e)
    return null
  }
}

export async function getJackpotExpectedValue(
  raceId: string,
  winProb: number,
): Promise<number> {
  try {
    const client = getClient()
    const config = await client.readContract({
      address: CONTRACT,
      abi: PET_RACING_ABI,
      functionName: 'getJackpotConfig',
    }) as { maxChanceBps: bigint; winnableBps: bigint; targetFee: bigint; eligibleJackpot: bigint }

    const payout = await previewPayouts(raceId)
    if (!payout || !payout.jackpotEligible) return 0

    const entryFeeWei = BigInt(payout.entryFee)
    const rollChance = Number(config.maxChanceBps) / 10000 *
      Math.min(Number(entryFeeWei), Number(config.targetFee)) / Number(config.targetFee)
    const jackpotPayout = weiToEth(
      ((config.eligibleJackpot * config.winnableBps) / 10000n).toString()
    )
    return winProb * rollChance * jackpotPayout
  } catch {
    return 0
  }
}

// lib/model/infer.ts
import type { ApiPetStats, RaceResult, PetSnapshot } from '@/types/racing'
import { decodeExtraParams, normalizeRarity, normalizeFaction, distanceBucket } from '@/lib/encode'

// Returns a 0–100 confidence score (how many of 15 slots are filled)
export function revealPercent(racesObserved: number): number {
  return Math.min(100, Math.round((racesObserved / 15) * 100))
}

// Map ApiPetStats → PetSnapshot (normalize types, decode conditions)
export function buildPetSnapshot(raw: ApiPetStats): PetSnapshot {
  const history: RaceResult[] = (raw.recentRaces ?? []).map((r) => {
    const { itemsMode, weather, faction } = decodeExtraParams(
      r.extraParamIds ?? [], r.extraParamVals ?? []
    )
    return {
      raceId: r.raceId,
      rank: r.rank,
      msFinishTime: r.msFinishTime,
      trackLength: r.trackLength,
      weather,
      itemsMode,
      factionStretchMix: faction ? { [faction]: 1 } : {},
      entryFee: r.entryFee ?? '0',
      fieldSize: r.fieldSize ?? 8,
    }
  })
  return {
    petId: raw.petId,
    rarity: normalizeRarity(raw.rarity),
    faction: normalizeFaction(raw.faction),
    gender: raw.gender as 'male' | 'female',
    elo: raw.elo,
    racesRun: raw.racesRun,
    history,
  }
}

// For a given pet and distance bucket: percentile finish time (lower = faster)
// Returns 0.5 if not enough data
export function distanceFinishPercentile(
  pet: PetSnapshot,
  bucket: 'short' | 'medium' | 'long',
): number {
  const relevant = pet.history.filter(r => distanceBucket(r.trackLength) === bucket)
  if (relevant.length < 2) return 0.5
  const times = relevant.map(r => r.msFinishTime).sort((a, b) => a - b)
  const myMedian = times[Math.floor(times.length / 2)]
  // Compare against all available times (self-relative for MVP)
  // Lower percentile = faster → return 1 - (rank/n)
  const rank = times.filter(t => t <= myMedian).length
  return 1 - rank / times.length
}

// Weather match bonus: how often does this pet perform above-median when weather matches?
// Returns delta vs baseline (-0.2 to +0.2)
export function weatherAffinityDelta(
  pet: PetSnapshot,
  weather: import('@/types/racing').Weather | null,
): number {
  if (!weather || pet.history.length < 3) return 0
  const withWeather = pet.history.filter(r => r.weather === weather)
  const without = pet.history.filter(r => r.weather !== weather && r.weather !== null)
  if (withWeather.length < 2 || without.length < 2) return 0
  const avgRankWith = withWeather.reduce((s, r) => s + r.rank, 0) / withWeather.length
  const avgRankWithout = without.reduce((s, r) => s + r.rank, 0) / without.length
  // Lower rank = better; positive delta means performs better in this weather
  const delta = (avgRankWithout - avgRankWith) / pet.history[0].fieldSize
  return Math.max(-0.2, Math.min(0.2, delta))
}

// Faction affinity: fraction of history races where pet's faction matched stretch
export function factionAffinityScore(pet: PetSnapshot): number {
  const withMatch = pet.history.filter(r => {
    const share = r.factionStretchMix[pet.faction] ?? 0
    return share > 0.25
  })
  if (pet.history.length === 0) return 0
  return withMatch.length / pet.history.length
}

// lib/model/handicap.ts
import type { PetSnapshot, RaceConditions, PayoutPreview, HandicapResult } from '@/types/racing'
import { rarityOrdinal, distanceBucket, weiToEth } from '@/lib/encode'
import {
  distanceFinishPercentile, weatherAffinityDelta,
  factionAffinityScore, revealPercent,
} from './infer'

const BETA = 1.5          // softmax temperature (calibrate in v2)
const RARITY_WEIGHT = 0.3 // bonus per rarity level (0-5 scale)

// Compute base strength score S_i for one pet
function baseStrength(pet: PetSnapshot, cond: RaceConditions): number {
  // 1. ELO z-score component (raw ELO, will be z-scored across field in handicap())
  let s = pet.elo

  // 2. Rarity prior
  s += rarityOrdinal(pet.rarity) * RARITY_WEIGHT * 100

  // 3. Distance condition bonus
  const bucket = distanceBucket(cond.trackLength)
  const distPerc = distanceFinishPercentile(pet, bucket)
  s += (distPerc - 0.5) * 200  // max ±100 pts

  // 4. Weather affinity
  const weatherDelta = weatherAffinityDelta(pet, cond.weather)
  s += weatherDelta * 200

  // 5. Faction affinity (if stretch mix known)
  const factionShare = cond.factionStretchMix[pet.faction] ?? 0
  if (factionShare > 0) {
    const affinity = factionAffinityScore(pet)
    s += factionShare * affinity * 100
  }

  return s
}

// Softmax over field scores → win probabilities
function softmax(scores: number[], beta: number): number[] {
  const max = Math.max(...scores)
  const exps = scores.map(s => Math.exp(beta * (s - max) / 1000))
  const sum = exps.reduce((a, b) => a + b, 0)
  return exps.map(e => e / sum)
}

// Simple Plackett–Luce approximation for rank distribution
function placketLuceRankDist(probs: number[], idx: number, fieldSize: number): number[] {
  const dist = new Array(fieldSize).fill(0)
  // P(rank=1) = p_i
  dist[0] = probs[idx]
  // P(rank=k) ≈ p_i * Σ_{subsets} ... approximate via scaled remainder
  // Simple approximation: each subsequent rank scaled by remaining probability mass
  let remaining = 1 - probs[idx]
  for (let k = 1; k < fieldSize; k++) {
    const others = fieldSize - k
    if (others <= 0) break
    dist[k] = probs[idx] * (remaining / (fieldSize - 1))
    remaining -= dist[k]
  }
  // Normalize
  const total = dist.reduce((a, b) => a + b, 0)
  return dist.map(d => d / total)
}

// Main handicap function
export function handicap(
  field: PetSnapshot[],
  cond: RaceConditions,
  payout: PayoutPreview | null,
): HandicapResult[] {
  const scores = field.map(pet => baseStrength(pet, cond))

  // Z-score the ELO part for better calibration
  const mean = scores.reduce((a, b) => a + b, 0) / scores.length
  const std = Math.sqrt(scores.map(s => (s - mean) ** 2).reduce((a, b) => a + b, 0) / scores.length) || 1

  const zScores = scores.map(s => (s - mean) / std)
  const winProbs = softmax(zScores, BETA)

  const entryFeeEth = payout ? weiToEth(payout.entryFee) : 0

  return field.map((pet, i) => {
    const winProb = winProbs[i]
    const fairOdds = winProb > 0 ? 1 / winProb : 99
    const rankDist = placketLuceRankDist(winProbs, i, field.length)

    // EV calculation: Σ P(rank=k) * payout_k - entryFee
    let evEnter: number | undefined
    if (payout && payout.payouts.length > 0) {
      const ev = rankDist.reduce((sum, prob, rank) => {
        const payoutForRank = payout.payouts.find(p => p.rank === rank + 1)
        return sum + prob * weiToEth(payoutForRank?.amount ?? '0')
      }, 0) - entryFeeEth
      evEnter = ev
    }

    // Value rating: how much edge vs implied pool odds
    const impliedOdds = payout && payout.payouts[0]
      ? weiToEth(payout.payouts[0].amount) / (entryFeeEth * field.length || 1)
      : fairOdds
    const valueRating = winProb * impliedOdds - 1  // positive = edge

    return {
      petId: pet.petId,
      winProb,
      fairOdds,
      rankDist,
      evEnter,
      valueRating,
      revealPct: revealPercent(pet.history.length),
    }
  })
}

'use client'
import Pusher from 'pusher-js'

let pusherInstance: Pusher | null = null

export function getPusher(): Pusher {
  if (pusherInstance) return pusherInstance
  pusherInstance = new Pusher(process.env.NEXT_PUBLIC_PUSHER_APP_KEY!, {
    cluster: process.env.NEXT_PUBLIC_PUSHER_CLUSTER ?? 'mt1',
  })
  return pusherInstance
}

export function subscribeToLobby(
  onRaceUpdated: (data: unknown) => void,
  onSnapshot: (data: unknown) => void,
): () => void {
  const p = getPusher()
  const channel = p.subscribe('racing.lobby')
  channel.bind('race-updated', onRaceUpdated)
  channel.bind('lobby-snapshot', onSnapshot)
  return () => {
    channel.unbind_all()
    p.unsubscribe('racing.lobby')
  }
}

export function subscribeToRace(
  raceId: string,
  onTick: (data: unknown) => void,
  onBroadcast: (data: unknown) => void,
): () => void {
  const p = getPusher()
  const channel = p.subscribe(`race-${raceId}`)
  channel.bind('tick-advanced', onTick)
  channel.bind('race-broadcast', onBroadcast)
  return () => {
    channel.unbind_all()
    p.unsubscribe(`race-${raceId}`)
  }
}

import type { ApiRaceSummary, RaceConditions } from '@/types/racing'

export const revalidate = 30

async function enrichRaces(races: ApiRaceSummary[]) {
  const openFull = races.filter(r => r.phase === 'OPEN' && (r.filledSlots ?? 0) > 0).slice(0, 12)
  
  // Collect all unique petIds across races for batch fetch
  const allPetIds = [...new Set(openFull.flatMap(r =>

  ))].filter(Boolean)
  
  let allSnapshots: ReturnType<typeof buildPetSnapshot>[] = []
  if (allPetIds.length > 0) {

      if (snapshots.length < 2) return { race, handicaps: [], pets: snapshots }
      
      const payout = await previewPayouts(race.raceId).catch(() => null)
      const { weather, itemsMode, faction } = decodeExtraParams(race.extraParamIds ?? [], race.extraParamVals ?? [])
      const cond: RaceConditions = {
        trackLength: race.trackLength,
        weather, itemsMode,
        factionStretchMix: faction ? { [faction]: 0.5 } : {},
      }
      const handicaps = handicap(snapshots, cond, payout)
      return { race, handicaps, pets: snapshots }
    } catch {
      return { race, handicaps: [], pets: [] }
    }
  }))
}

export default async function LobbyPage() {
  let races: ApiRaceSummary[] = []
  try {
    const data = await fetchLobbySync()
    races = data.races ?? []
  } catch (e) {
    console.error('Lobby fetch failed:', e)
  }

  const open = races.filter(r => r.phase === 'OPEN')
  const enriched = await enrichRaces(open)
  const edgeRaces = enriched.filter(r => r.handicaps.some(h => (h.evEnter ?? -1) > 0))

  return (
    <div>
      <div className="mb-6">
        <h1 className="text-2xl font-bold text-slate-100">Open Races</h1>
        <div className="flex items-center gap-3 mt-1">
          <p className="text-slate-500 text-sm">{open.length} open</p>
          {edgeRaces.length > 0 && (
            <span className="text-xs text-emerald-400 font-medium">
              {edgeRaces.length} with EDGE
            </span>
          )}
          <LiveLobby initialCount={open.length} />
        </div>
      </div>

      {open.length === 0 ? (
        <div className="text-center py-16 text-slate-500">No open races right now.</div>
      ) : (
        <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4">
          {enriched.map(({ race, handicaps, pets }) => (
            <RaceCard key={race.raceId} race={race} handicaps={handicaps} pets={pets} />
          ))}
        </div>
      )}
    </div>
  )
}

// app/api/ai-read/route.ts
import { NextResponse } from 'next/server'
import Anthropic from '@anthropic-ai/sdk'
import type { HandicapResult, PetSnapshot } from '@/types/racing'

const client = new Anthropic()

export async function POST(req: Request) {
  const { pets, handicaps, conditions } = await req.json() as {
    pets: PetSnapshot[]
    handicaps: HandicapResult[]
    conditions: { trackLength: number; weather: string | null }
  }

  const sorted = [...handicaps].sort((a, b) => b.winProb - a.winProb).slice(0, 3)
  const descriptions = sorted.map(h => {
    const pet = pets.find(p => p.petId === h.petId)
    return `Pet #${h.petId} (${pet?.rarity}, ${pet?.faction}): ${(h.winProb * 100).toFixed(1)}% win prob, ${h.revealPct}% profile revealed`
  }).join('\n')

  const prompt = `Race conditions: ${conditions.trackLength}m track, weather: ${conditions.weather ?? 'unknown'}
Top 3 contenders:
${descriptions}

In 2 sentences, explain which pet looks best and why, considering what we know. Be specific, mention uncertainty where profile is sparse.`

  try {
    const message = await client.messages.create({
      model: 'claude-haiku-4-5-20251001',
      max_tokens: 150,
      messages: [{ role: 'user', content: prompt }],
    })
    const text = message.content[0].type === 'text' ? message.content[0].text : ''
    return NextResponse.json({ text })
  } catch (e) {
    return NextResponse.json({ error: String(e) }, { status: 500 })
  }
}
